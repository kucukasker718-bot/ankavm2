WiseCP & WHMCS Entegrasyon Modülü

WiseCP Server sanallaştırma modülü entegrasyonu kod blokları

WiseCP veya WHMCS faturalandırma yazılımınızdan AnkaVM API Gateway sunucunuza bağlanmak için aşağıdaki PHP modülünü kullanın. Bu dosya WiseCP sunucunuzda aşağıdaki dizine yüklenmelidir:

/cpanel/modules/Servers/AnkaVM/AnkaVM.php
Modül Kaynak Kodu (PHP-Hook): 
<?php
/**
 * AnkaVM Server Virtualization Module - WiseCP & WHMCS
 * Place at: /cpanel/modules/Servers/AnkaVM/AnkaVM.php
 */
class AnkaVM {
    private $apiUrl;
    private $apiKey;

    public function __construct($apiUrl, $apiKey) {
        $this->apiUrl = rtrim($apiUrl, '/');
        $this->apiKey = $apiKey;
    }

    private function call($endpoint, $method = 'GET', $data = null) {
        $ch = curl_init();
        $url = $this->apiUrl . '/' . ltrim($endpoint, '/');
        $headers = [
            'X-API-Key: ' . $this->apiKey,
            'Content-Type: application/json'
        ];
        curl_setopt($ch, CURLOPT_URL, $url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_HTTPHEADER, $headers);
        curl_setopt($ch, CURLOPT_CUSTOMREQUEST, $method);
        if ($data) curl_setopt($ch, CURLOPT_POSTFIELDS, json_encode($data));
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false);
        $response = curl_exec($ch);
        curl_close($ch);
        return json_decode($response, true);
    }

    public function createServer($vmName, $cpu, $ram, $disk, $template, $rootPassword) {
        return $this->call('/api/vms', 'POST', [
            'name' => $vmName,
            'cpu' => $cpu,
            'ram_mb' => $ram,
            'disk_gb' => $disk,
            'os_template' => $template,
            'root_password' => $rootPassword
        ]);
    }
}
